  255.     function checkForm($fields)

  257.         // First check that the password and confirmpassword match
    258. 


  258.         if($fields['password'] != $fields['confirmpassword']){
    259. 


  259.             return array("message" => AUCTION_NONMATCHING_PASSWORDS, "field" => "password");

  264.             unset($fields['confirmpassword']);
    265.

  266.  
    267. 


  267.         foreach($fields as $k => $v){

  268.             $result = true; 
    269. 


  269.             // If there is no method defined for this field, skip the validation
    270. 


  270.             if(method_exists($this, "_check" . ucfirst($k))){ 
    271. 


  271.                 // Each method outputs an error message if the input is invalid
    272. 


  272.                 // The variable $reserve must be checked against $startingprice
    273. 


  273.                 if($k == "reserve"){
    274. 


  274.                     $result = $this->{"_check" . ucfirst($k)}($v, $fields['startprice']);
    275. 


  275.                 }else{
    276. 


  276.                     $result = $this->{"_check" . ucfirst($k)}($v);
    277. 


  277.                 }
    278. 


  278.             }elseif(($fields['formdata'] == "register" OR $fields['formdata'] == "update") && $v == ""){
    279. 


  279.                 return array("message" => AUCTION_ALL_FIELDS_REQUIRED, "field" => $k);
    280. 


  280.             }elseif($fields['formdata'] == "submission" && $v == "" && $k != "reserve"){
    281. 


  281.                 return array("message" => AUCTION_ALL_FIELDS_REQUIRED, "field" => $k);            
    282. 


  282.             }
    283. 


  283.             if($result !== true){
    284. 


  284.                 return $result;
    285. 


  285.             }
    286.

  287.         return true;
