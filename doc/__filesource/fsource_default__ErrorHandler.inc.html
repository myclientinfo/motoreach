  154.     function load_ini( $inifile ){
    155. 


  155.         if ( !file_exists($inifile) ){
    156. 


  156.             return FALSE;
    157. 


  157.         }
    158. 


  158.         $ini = parse_ini_file($inifile, TRUE);
    159. 


  159.         foreach ( $ini as $section => $setting ){
    160. 


  160.             if ( is_array($setting) ){
    161. 


  161.                 // process SECTION
    162. 


  162.                 foreach ( $setting as $flag => $value ){
    163. 


  163.                     // if flag or value can be a constant, use that constant intead
    164. 


  164.                     // greedy method ~> I'm afraid it should be changed shortly
    165. 


  165.                     if ( defined($flag) ){
    166. 


  166.                         $flag = constant($flag);
    167. 


  167.                     }
    168. 


  168.                     if ( in_array($section, $this->_SECTION) ){
    169. 


  169.                         // built-in SECTION directives
    170. 


  170.                         if ( !strcasecmp($flag, 'level') ){
    171. 


  171.                             $this->set($section, $this->_intlevel($value));
    172. 


  172.                         } else {
    173. 


  173.                              call_user_func(array(&$this, 'set_'.$section), $flag, $value); // $this->{"set_$section"}($flag, $value);
    174. 


  174.                         }
    175. 


  175.                     } else {
    176. 


  176.                         // not built-in SECTION directives ~> ADLTLOG file-registering
    177. 


  177.                         $this->set_altdlog($section, $flag, $value);
    178. 


  178.                     }
    179. 


  179.                 }
    180. 


  180.             } else {
    181. 


  181.                 // common GLOBAL settings
    182. 


  182.                 switch ( $flag = strtoupper($section) ){
    183. 


  183.                 case 'LEVEL' :
    184. 


  184.                     $this->set('DEFAULT', $this->_intlevel($setting));
    185. 


  185.                     break;
    186. 


  186.                 case 'CUSTOM':
    187. 


  187.                     $this->CUSTOM  = is_callable($setting) ? $setting : NULL;
    188. 


  188.                     break;
    189. 


  189.                 case 'LOCKED': case 'SILENT':
    190. 


  190.                     $this->{$flag} = (bool)$setting;
    191. 


  191.                 default:
    192. 


  192.                     break;
    193. 


  193.                 }
    194. 


  194.             }
    195. 


  195.         }
    196. 


  196.         // if global LEVEL is not specified, try to get from php.ini
    197. 


  197.         
    198. 


  198.         if ( !isset($this->_LEVEL['DEFAULT']) ){
    199. 


  199.             $this->set('DEFAULT', ini_get('error_reporting'));
    200. 


  200.         }
    201. 


  201.         // lock ErrorHandler configuration properties
    202. 


  202.         if ( !empty($this->LOCKED) ){
    203. 


  203.             return $this->lock();
    204. 


  204.         }
    205. 


  205.         return TRUE;
    206. 


  206.     }

  232.     function set( $section, $level ){
    233. 


  233.         static $prev_level = null;
    234. 


  234.  
    235. 


  235.         if ( defined('ERRORHANDLER_LOCKED') || (!in_array($section, $this->_SECTION) && strcmp($section, 'DEFAULT')) ){
    236. 


  236.             return FALSE;
    237. 


  237.         }
    238. 


  238.         // "quick switch" (ON/OFF), isset(...) indicates that it can be used after load_ini() only
    239. 


  239.         if ( is_bool($level) && isset($this->_LEVEL[$section]) ){
    240. 


  240.             if ( $level ) {
    241. 


  241.         // turn ON report (specified in $section)
    242. 


  242. // attempt to  restore previous level, otherwise use DEFAULT
    243. 


  243.                 
    244. 


  244.                 $prev_level[$section]   = $this->_LEVEL[$section];
    245. 


  245.                 $this->_LEVEL[$section] = !empty($prev_level[$section])?$prev_level[$section]:$this->_LEVEL['DEFAULT'];
    246. 


  246.             }
    247. 


  247.             else if ( !empty($this->_LEVEL[$section]) ){
    248. 


  248.                 // turn OFF report (specified in $section) if possible (level > 0)
    249. 


  249.                 $prev_level[$section]   = $this->_LEVEL[$section];
    250. 


  250.                 $this->_LEVEL[$section] = 0;
    251. 


  251.             }
    252. 


  252.         } else { // general purpose switch / change the actual report level
    253. 


  253.             $level = $this->_intlevel($level);
    254. 


  254.             $prev_level[$section] = $this->_LEVEL[$section];
    255. 


  255.             $this->_LEVEL['ALL'] |= $this->_LEVEL[$section] = $level;
    256. 


  256.             if ( strcasecmp($section, 'DEFAULT') == 0 ){
    257. 


  257.                 foreach ( $this->_SECTION as $s ){
    258. 


  258.                     if ( empty($this->_LEVEL[$s]) ){
    259. 


  259.             # TODO: isset -- empty -- === 0 mizeria mi van, ha 0-ra van allitva, akkor felulirja???
    260. 


  260. # pedig lehet, hogy direkt ki akarta kapcsolni
    261. 


  261.                         
    262. 


  262.                         $this->_LEVEL[$s] = $this->_LEVEL['DEFAULT'];
    263. 


  263.                     }
    264. 


  264.                 }
    265. 


  265.             }
    266. 


  266.             return $prev_level[$section];
    267. 


  267.         }
    268. 


  268.     }

  300.         if ( defined('ERRORHANDLER_LOCKED') ){
    301. 


  301.             return FALSE;

  315.                     $this->_CONTEXT['exclude'] =
    316. 


  316.                         array_merge($this->_CONTEXT['exclude'], preg_split('/[\W]+/', $arg, -1, PREG_SPLIT_NO_EMPTY));

  317.                 }
    318. 


  318.                 else if ( is_array($arg) ){
    319. 


  319.                     $this->_CONTEXT['exclude'] = array_merge($this->_CONTEXT['exclude'], $arg);

  322.         default:

  443.     function _context( $level, &$var_context, $source ){
    444.

    445. 


  445.         if ( !($this->_LEVEL['CONTEXT'] & $level) ){
    446.

  448.         $context   = sprintf("`--> CONTEXT REPORT (strict: %s, filtered: %s)\n",
    449.

  450.         $var_names = array_keys($var_context);

  493.     function _handle_error( $level, $err_str, $file_name, $line_no, $context ){
    494. 


  494.         global $HTTP_SERVER_VARS;
    495. 


  495.  
    496. 


  496.         // if this error should be trapped

    528.

  531.     }

  642.     function _source( $level, $file_name, $line_no ){
    643. 


  643.         // if SOURCE report should be generated
    644. 


  644.         if ( ($this->_LEVEL['SOURCE'] & $level) == 0 ){
    645. 


  645.             return '';
    646. 


  646.         }
    647. 


  647.         $header = sprintf("`--> SOURCE REPORT from %s around line %d", $file_name, $line_no);
    648. 


  648.         if ($line_no == 0){
    649. 


  649.             return $header."\n";
    650. 


  650.         }
    651. 


  651.         $file = file($file_name, 1);
    652. 


  652.         while ( $this->_SOURCE['block'] ){ // in fact, it's not a really loop (exit point at the end)
    653. 


  653.             for ( $i = $line_no-1, $d = -1, $block = 0; ($d == -1 && $i) || ($block > 0 && $i < count($file)); $i += $d ){
    654. 


  654.                 if ( preg_match('/(?: function\s+\w+\s*\(.*\) | (?:(?:else)?if|switch|while|declare) \s*\(.+\) | else | do
    655. 


  655.                     | for\s*\(.*;.*;.*\) | foreach\s*\(.* as\s+\$\w+(?:\s*=\>\s*\$\w+)?\s*\) ) \s*{ /Sxi', $file[$i]) ){
    656. 


  656.                     if (  ++$block > 0 ){
    657. 


  657.                         $d = +1;
    658. 


  658.                         $j = $i;
    659. 


  659.                         $i = $line_no-1;
    660. 


  660.                     }
    661. 


  661.                 } else {
    662. 


  662.                     preg_match_all('!(?'.'>[^/#}]*) (?(?=.) ( (?://) | [}#] ) )!x', $file[$i], $match);
    663. 


  663.                     foreach( $match[1] as $m ){
    664. 


  664.                         if ( $m == '}' ){ $block--; } else { break; }
    665. 


  665.                     }
    666. 


  666.                 }
    667. 


  667.             }
    668. 


  668.             // not found starting block
    669. 


  669.             if ( $d == -1 && !$i ){
    670. 


  670.                 break; // that's why we needed fake loop above!
    671. 


  671.             }
    672. 


  672.             $source = join('', array_slice($file, $j, $i-$j));
    673. 


  673.             return sprintf("%s (%d-%d)\n%s", $header, $j, $i, $this->_phpscope($source));
    674. 


  674.         }
    675. 


  675.         $context = $this->_SOURCE['lines'] >= 0 ? $this->_SOURCE['lines'] : 1;
    676. 


  676.         $offset  = $line_no-1 >= $context ? $line_no-1 - $context : 0;
    677. 


  677.         $count   = 2*$context+1;
    678. 


  678.  
    679. 


  679.         $source  = array_slice($file, $offset, $count);
    680. 


  680.         $count   = count($source);
    681. 


  681.         $source  = join('', $source);
    682. 


  682.         return sprintf("%s (%d-%d)\n%s", $header, $offset+1, $offset+$count, $this->_phpscope($source));
    683. 


  683.     }

  742.     function _sprint_var( $var, $name, $level ){
    743. 


  743.         $pad  = str_pad('', 2*$level, ' ');
    744. 


  744.  
    745. 


  745.         if ( is_array($var) || is_object($var) ) {
    746. 


  746.             if ( is_object($var) ) {
    747. 


  747.                 $buffer = sprintf("$pad%s is an object(%s) ", $level ? "'$name'" : $name, get_class($var));
    748. 


  748.                 $var    = get_object_vars($var);
    749. 


  749.             } else {
    750. 


  750.                 $buffer = sprintf("$pad%s is an array[%d] ",  $level ? "'$name'" : $name, count($var));
    751. 


  751.             }
    752. 


  752.             if($level < $this->_CONTEXT['depth']) {
    753. 


  753.                 $buffer .= "{\n";
    754.
